Skript na přidání. Klíč nutno poté zkontrolovat!

mkdir ~/.ssh 2>/dev/null; \
wget -q --output-document=- http://lukas.zapletalovi.com/assets/static/authorized_keys.txt >> \
~/.ssh/authorized_keys && \
chmod 755 ~/.ssh && \
chmod 600 ~/.ssh/authorized_keys
